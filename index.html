How to use ?

<?php

date_default_timezone_set('Europe/Paris');
include 'PhpICS/src/index.php';

$icalc = ICS::open('test.ics');
// $icalc = ICS::load(file_get_content('test.ics'));

foreach( $icalc as $event ) {
  echo $event->getDateStart('Y-m-d H:i:s'), ' - ', $event->getSummary(), '<br />';
}

echo '<pre>', $icalc, '</pre>';
